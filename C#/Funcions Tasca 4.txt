using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication2
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] array = { 1, 2, 33, 4, 5, 6 };
            int gran;
            gran = maxim(array);
            Console.WriteLine("El nombre mes gran de l'array es " + gran);
        }
        public static int maxim(int[] x)
        {
            int maxim = x[0], llargada = x.Length;
            for (int compt = 1; compt < llargada; compt++)
            {
                if (x[compt] > maxim) maxim = x[compt];
            }
            return maxim;
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication2
{
    class Program
    {
        public static bool bucle = false;
        static void Main(string[] args)
        {
            int[] array = { 22, 45, 67, -3 };
            while (bucle == false)
            {
                Console.WriteLine("\nAra l'array conté els nombres següents:\n" + array[0] + "\n" + array[1] + "\n" + array[2] + "\n" + array[3]);
                referencia(ref array);
            }
        }
        public static int referencia(ref int[] array)
        {
            char line;
            Console.WriteLine("\nVols canviar algun nombre de l'array? Y/N: ");
            line = Console.ReadKey().KeyChar;
            if (line == 'y')
            {
                Console.WriteLine("\n\nQuin nombre vols canviar? Prem 1, 2, 3 o 4: ");
                line = Console.ReadKey().KeyChar;
                if (line == '1')
                {
                    Console.WriteLine("\n\nQuin vols que sigui el nou nombre?");
                    array[0] = Convert.ToInt32(Console.ReadLine());
                }
                else if (line == '2')
                {
                    Console.WriteLine("\n\nQuin vols que sigui el nou nombre?");
                    array[1] = Convert.ToInt32(Console.ReadLine());
                }
                else if (line == '3')
                {
                    Console.WriteLine("\n\nQuin vols que sigui el nou nombre?");
                    array[2] = Convert.ToInt32(Console.ReadLine());
                }
                else if (line == '4')
                {
                    Console.WriteLine("\n\nQuin vols que sigui el nou nombre?");
                    array[3] = Convert.ToInt32(Console.ReadLine());
                }
                else if (line != '1' & line != '2' & line != '3' & line != '4')
                {
                    Console.WriteLine("\n\nNombre no valid.");
                }
            }
            else bucle = true;
            return 0;
        }
    }
}




Exercici Anys Traspas C#:
-------------------------

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication2
{
    class Program
    {
        static void Main(string[] args)
        {
            int any1 = 1582, any2 = 2017;
            AnysTraspas(any1, any2);
        }
        public static int AnysTraspas(int any1, int any2)
        {
            bool traspas = false;
            for (int compt = any1; compt <= any2; compt++)
            {
                traspas = EsAnyTraspas(compt);
                if (traspas == true) Console.WriteLine(compt);
            }
            return 0;
        }
        public static bool EsAnyTraspas(int compt)
        {
            if ((compt % 400 == 0) | ((compt % 4 == 0) & (compt % 100 != 0))) return true;
            else return false;
        }
    }
}
